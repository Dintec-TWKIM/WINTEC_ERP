		/// <summary>
		/// 컨트롤들의 캡션을 데이터 사전을 이용하여 설정한다.
		/// </summary>
		private void InitControl()
		{
			foreach(Control ctr in this.Controls)
			{
				if(ctr is Panel)
				{
					foreach(Control ctrl in ((Panel)ctr).Controls)
					{
						if(ctrl is Panel)
						{
							foreach(Control ctrls in ((Panel)ctrl).Controls)
							{
								if(ctrls is Label)
									((Label)ctrls).Text = GetDataDictionaryItem(DataDictionaryTypes.SA, (string)((Label)ctrls).Tag);
							}
						}
					}
				}
			}

			bpNm_Sl.Tag ="";
		}

		#region -> GetDDItem

		private string GetDDItem(params string[] colName)
		{
			string temp = "";
			
			for(int i = 0; i < colName.Length; i++)
			{
				switch(colName[i])		// DataView 의 컬럼이름
				{
					case "S":			// S
						temp = temp + " + " +  "S";
						break;
					case "NO_GIR":		
						temp = temp + " + " + this.GetDataDictionaryItem("SA","NO_GIR");
						break;
					case "DT_GIR":		
						temp = temp + " + " + this.GetDataDictionaryItem("SA","DT_GIR");
						break;
					case "NM_PARTNER":		
						temp = temp + " + " + this.GetDataDictionaryItem("SA","CD_PARTNER");
						break;
					case "NM_PLANT":		
						temp = temp + " + " + this.GetDataDictionaryItem("SA","PLANT_GIR");
						break;
					case "NM_RETURN":			
						temp = temp + " + " + this.GetDataDictionaryItem("SA","FG_GI");
						break;
					case "CD_ITEM":		// 품번
						temp = temp + " + " + this.GetDataDictionaryItem("SA","CD_ITEM");
						break;
					case "NM_ITEM":		// 품명
						temp = temp + " + " + this.GetDataDictionaryItem("SA","NM_ITEM");
						break;
					case "STND_ITEM":	// 규격
						temp = temp + " + " + this.GetDataDictionaryItem("SA","STND_ITEM");
						break;
					case "UNIT_IM":		// 단위
					case "UNIT":		// 단위
						temp = temp + " + " + this.GetDataDictionaryItem("SA","UNIT");
						break;
					case "NM_SL":		// 창고명
						temp = temp + " + " + this.GetDataDictionaryItem("SA","SL_GIR");
						break;
					case "NM_BUSI":		
						temp = temp + " + " + this.GetDataDictionaryItem("SA","TP_BUSI");
						break;
					case "QT_GIR":		// 수배의뢰량
						temp = temp + " + " + this.GetDataDictionaryItem("SA","QT_GIR");
						break;
					case "QT_GI":		// 환종
						temp = temp + " + " + this.GetDataDictionaryItem("SA","QT_GI");
						break;
					case "NM_EXCH":			// 환종단가
						temp = temp + " + " + this.GetDataDictionaryItem("SA","CD_EXCH");
						break;
					case "UM":		// 금액
						temp = temp + " + " + this.GetDataDictionaryItem("SA","PRICE");
						break;					
					case "AM_GIR":		// 부가세
						temp = temp + " + " + this.GetDataDictionaryItem("SA","AM");
						break;
					case "AM_GIRAMT":		// 재고단위
						temp = temp + " + " + this.GetDataDictionaryItem("SA","AM_WON");
						break;
					case "NM_QTIOTP":		// 관련수주번호
						temp = temp + " + " + this.GetDataDictionaryItem("SA","TP_GI");
						break;
					case "NM_SALEGRP":// 납품처
						temp = temp + " + " + this.GetDataDictionaryItem("SA","CD_SALEGRP");
						break;
					case "DT_DUEDATE":		// 환종
						temp = temp + " + " + this.GetDataDictionaryItem("SA","DT_DUEDATE");
						break;				
					case "NO_SO":		// 환종
						temp = temp + " + " + this.GetDataDictionaryItem("SA","NO_SO");
						break;
					case "SEQ_SO":			// 환종단가
						temp = temp + " + " + this.GetDataDictionaryItem("SA","SEQ");
						break;
					case "TP_GI":			// 출하형태
						temp = temp + " + " + this.GetDataDictionaryItem("SA","TP_GI");
						break;
					case "CD_SALEGRP":			// 환종단가
						temp = temp + " + " + this.GetDataDictionaryItem("SA","CD_SALEGRP");
						break;
					case "CD_PARTNER":			// 환종단가
						temp = temp + " + " + this.GetDataDictionaryItem("SA","CD_PARTNER");
						break;
					case "LN_PARTNER":			// 환종단가
						temp = temp + " + " + this.GetDataDictionaryItem("SA","NM_PARTNER");
						break;
					case "NO_PROJECT":			// 환종단가
						temp = temp + " + " + this.GetDataDictionaryItem("SA","NO_PROJECT");
						break;
					case "NM_PROJECT":			// 환종단가
						temp = temp + " + " + this.GetDataDictionaryItem("SA","NM_PROJECT");
						break;
                    case "NM_TP_GI":			// 출하형태
                        temp = temp + " + " + this.GetDataDictionaryItem("SA", "TP_GI");
                        break;
					default :
						break;

				}
			}
			
			if(temp == "")
				return "";
			else
				return temp.Substring(3,temp.Length-3);
		}

		private DataTable SelectedTabNO_GIR()
		{
			DataTable ldt_report = gdv_LineNO_GIR.Table.Clone();

			try
			{			
				DataTable ldt_Selected = _flexM_1.DataTable;

				DataRow[] ldr_Args = ldt_Selected.Select("S ='Y'");
				if( ldr_Args != null && ldr_Args.Length >0)
				{
					for( int i= 0 ; i < ldr_Args.Length ; i++)
					{
						string ls_filter = "NO_GIR = '" +ldr_Args[i]["NO_GIR"].ToString().Trim()+ "'";	

						DataRow[] ldr_Selectd = gdv_LineNO_GIR.Table.Select(ls_filter,"",DataViewRowState.Unchanged);
						if( ldr_Selectd != null && ldr_Selectd.Length >0)
						{
							for(int j= 0 ; j < ldr_Selectd.Length ; j++)
							{
								ldt_report.ImportRow(ldr_Selectd[j]);
							}
						}						
					}
				}
			}
			catch(Exception ex)
			{
				throw ex;
			}
			return ldt_report;
		}

		/// <summary>
		/// 릴레이션
		/// </summary>
		/// <param name="_dtH"></param>
		/// <param name="_dtD"></param>
		/// <param name="r_name"></param>
		/// <returns></returns>
		private DataTable MakeRelation(FlexGrid _gM, FlexGrid _gD, string r_name)
		{
		
			DataTable _dtM = _gM.DataTable.Copy();
			DataTable _dtD = _gD.DataTable.Copy();

			DataTable _dtCp = new DataTable();
			_dtCp = _dtD.Clone();

			for(int i = 0; i < _dtM.Columns.Count; i++)
			{
				if(_dtCp.Columns.IndexOf(_dtM.Columns[i].ColumnName) < 0)
				{
					_dtCp.Columns.Add(_dtM.Columns[i].ColumnName);
				}
			}
			_dtCp.Clear();

            DataSet ds = new DataSet();
            ds.Tables.Add(_dtM);
            ds.Tables.Add(_dtD);

			DataRelation myDataRelation = new DataRelation("parent2Child", _dtM.Columns[r_name], _dtD.Columns[r_name], false);

            ds.Relations.Add(myDataRelation);

			foreach (DataRow _drD in _dtD.Rows)
			{
				DataRow dr = _dtCp.NewRow();
				dr.ItemArray = _drD.ItemArray;

				foreach(DataRow _drM in _drD.GetParentRows(myDataRelation))
				{
					for(int i = 0; i < _dtM.Columns.Count; i++)
					{
						if(_dtCp.Columns.IndexOf(_dtM.Columns[i].ColumnName) >= 0)
						{
							dr[_dtM.Columns[i].ColumnName] = _drM[_dtM.Columns[i].ColumnName];
						}
					}
				}

				_dtCp.Rows.Add(dr);
			}

			int c1 = _gM.FindRow("Y",_gM.Rows.Fixed, _gM.Cols["S"].Index, false, true, true);
			if(c1 > 0)
			{
				DataRow [] _dr = _dtCp.Select("S = 'Y'");
				DataTable _dt = _dtCp.Clone();

				for(int i = 0; i < _dr.Length; i++)
				{
					_dt.ImportRow(_dr[i]);
				}
				_dtCp.Clear();
				_dtCp = _dt;
			}

			return _dtCp;
		}

		{
			if(e.KeyData.ToString() == "Enter" || e.KeyData.ToString() == "Down")
			{			
				SendKeys.SendWait("{TAB}");
			}

		}

		}
